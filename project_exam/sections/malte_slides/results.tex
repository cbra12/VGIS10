\section{Conclusion}
\begin{frame}{Conclusion}{}
    \begin{block}{Conclusion}
        \begin{itemize}
            \item It is possible to increase the PSNR performance on images from the same environment
            \item Minor visual difference on the images we have tested on
        \end{itemize}
    \end{block}
